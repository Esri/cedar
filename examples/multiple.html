Multiple Charts

<html>
<head>
  <meta charset=utf-8 />
  <title></title>
  <meta name='viewport' content='initial-scale=1,maximum-scale=1,user-scalable=no' />

  <!-- Load D3 -->
  <script type="text/javascript" src="http://square.github.io/crossfilter/d3.v3.min.js"></script>
  <script type="text/javascript" src="http://trifacta.github.io/vega/vega.js"></script>

  <!-- Load Cedar -->
  <script type="text/javascript" src="cedar.js"></script>

</head>
<body>

<div class="row">
  <div class="col-lg-12" id="chart"></div>
</div>

<script>

  var dataset = {
    "url":"http://maps2.dcgis.dc.gov/dcgis/rest/services/DCGIS_DATA/Education_WebMercator/MapServer/5",
    "mappings":{
      "group": {"field":"CAPITALGAINS","label":"Capital Gains"},
      "count": {"field":"CAPACITY","label":"Student Capacity"}
    }
  };

  var chart1 = new Cedar({"specification":"../../../cedar/data/templates/bar.json", "dataset": dataset});

  chart1.show({
    elementId: "#chart1",
  });  

  var chart2 = new Cedar({"specification":"../../../cedar/data/templates/pie.json", "dataset": dataset});

  chart2.show({
    elementId: "#chart2",
  });    

</script>
</body>
</html>
